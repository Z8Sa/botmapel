// ================= DATABASE MANAGEMENT =================
class Database {
    constructor() {
        this.db = this.load();
    }

    load() {
        if (fs.existsSync(DB_PATH)) {
            try {
                const data = fs.readFileSync(DB_PATH, "utf-8");
                if (!data.trim()) return this.createDefault();
                return JSON.parse(data);
            } catch (err) {
                console.log("Error loading database, creating new one...");
                return this.createDefault();
            }
        }
        return this.createDefault();
    }

    createDefault() {
        const defaultDB = { semester1: {}, semester2: {} };
        for (const m of MAPEL) {
            defaultDB.semester1[m] = { materi: [], tugas: [] };
            defaultDB.semester2[m] = { materi: [], tugas: [] };
        }
        this.save(defaultDB);
        return defaultDB;
    }

    save(data = this.db) {
        fs.writeFileSync(DB_PATH, JSON.stringify(data, null, 2), "utf-8");
        this.db = data;
    }

    getMapel(sem, mapel) {
        if (!this.db[sem][mapel]) {
            this.db[sem][mapel] = { materi: [], tugas: [] };
        }
        return this.db[sem][mapel];
    }

    getAllMapel(semester) {
        return Object.keys(this.db[semester]);
    }

    getInfo() {
        let info = "📚 INFORMASI DATABASE BOT:\n\n";
        
        info += "SEMESTER 1:\n";
        for (const mapel in this.db.semester1) {
            const materiCount = this.db.semester1[mapel].materi.length;
            const tugasCount = this.db.semester1[mapel].tugas.length;
            if (materiCount > 0 || tugasCount > 0) {
                info += `• ${mapel}: ${materiCount} materi, ${tugasCount} tugas\n`;
            }
        }
        
        info += "\nSEMESTER 2:\n";
        for (const mapel in this.db.semester2) {
            const materiCount = this.db.semester2[mapel].materi.length;
            const tugasCount = this.db.semester2[mapel].tugas.length;
            if (materiCount > 0 || tugasCount > 0) {
                info += `• ${mapel}: ${materiCount} materi, ${tugasCount} tugas\n`;
            }
        }
        
        return info;
    }

    search(query) {
        const results = [];
        const lowerQuery = query.toLowerCase();
        
        for (const sem of ["semester1", "semester2"]) {
            for (const mapel in this.db[sem]) {
                const materiList = this.db[sem][mapel].materi.filter(m => 
                    m.toLowerCase().includes(lowerQuery)
                );
                const tugasList = this.db[sem][mapel].tugas.filter(t => 
                    t.toLowerCase().includes(lowerQuery)
                );
                
                if (materiList.length > 0 || tugasList.length > 0) {
                    results.push({
                        semester: sem === "semester1" ? "Semester 1" : "Semester 2",
                        mapel: mapel,
                        materi: materiList,
                        tugas: tugasList
                    });
                }
            }
        }
        return results;
    }
}

class ImageGenerator {
    static async generateMapelImage(mapel, semester, data) {
        const width = 800;
        const height = 800;
        const canvas = createCanvas(width, height);
        const ctx = canvas.getContext("2d");

        const gradient = ctx.createLinearGradient(0, 0, 0, height);
        gradient.addColorStop(0, "#1a1a2e");
        gradient.addColorStop(1, "#16213e");
        ctx.fillStyle = gradient;
        ctx.fillRect(0, 0, width, height);

        ctx.strokeStyle = "#0f3460";
        ctx.lineWidth = 5;
        ctx.strokeRect(10, 10, width - 20, height - 20);

        ctx.fillStyle = "#e94560";
        ctx.font = "bold 32px sans-serif";
        ctx.fillText(`📘 ${mapel.toUpperCase()}`, 30, 70);
        
        ctx.fillStyle = "#ffd93d";
        ctx.font = "20px sans-serif";
        ctx.fillText(semester === "semester1" ? "SEMESTER 1" : "SEMESTER 2", 30, 115);

        ctx.beginPath();
        ctx.strokeStyle = "#533483";
        ctx.lineWidth = 2;
        ctx.moveTo(30, 135);
        ctx.lineTo(width - 30, 135);
        ctx.stroke();

        ctx.fillStyle = "#00b4d8";
        ctx.font = "bold 24px sans-serif";
        ctx.fillText("📖 MATERI", 30, 180);

        ctx.fillStyle = "#ffffff";
        ctx.font = "16px sans-serif";
        let y = 220;
        
        if (data.materi.length === 0) {
            ctx.fillStyle = "#888888";
            ctx.fillText("   ✨ Belum ada materi", 30, y);
            y += 30;
        } else {
            data.materi.forEach((m, i) => {
                const text = `   ${i + 1}. ${m.length > 60 ? m.slice(0, 57) + "..." : m}`;
                ctx.fillStyle = "#cccccc";
                ctx.fillText(text, 30, y);
                y += 30;
                if (y > height - 150) return;
            });
        }

        y += 20;
        ctx.fillStyle = "#ff6b6b";
        ctx.font = "bold 24px sans-serif";
        ctx.fillText("📝 TUGAS", 30, y);
        y += 40;

        ctx.fillStyle = "#ffffff";
        ctx.font = "16px sans-serif";
        
        if (data.tugas.length === 0) {
            ctx.fillStyle = "#888888";
            ctx.fillText("   ✨ Belum ada tugas", 30, y);
        } else {
            data.tugas.forEach((t, i) => {
                const text = `   ${i + 1}. ${t.length > 60 ? t.slice(0, 57) + "..." : t}`;
                ctx.fillStyle = "#ffd93d";
                ctx.fillText(text, 30, y);
                y += 30;
                if (y > height - 50) return;
            });
        }

        ctx.fillStyle = "#533483";
        ctx.font = "12px sans-serif";
        ctx.fillText("DEVELOPER : RAFI", width - 150, height - 20);

        return canvas.toBuffer();
    }
}

// ================= AI SERVICE (DIPERBAIKI) =================
class AIService {
    static async ask(question, database) {
        const dbKeywords = ["materi", "tugas", "database", "mapel", "pelajaran", "semester", "konten", "isi bot", "apa", "ada", "berapa", "lihat", "tampilkan"];
        const isAskingDatabase = dbKeywords.some(keyword => question.toLowerCase().includes(keyword));
        
        let dbContext = "";
        
        if (isAskingDatabase) {
            const searchResults = database.search(question);
            
            if (searchResults.length > 0) {
                dbContext = "\n\n[DATA DARI DATABASE BOT]\n";
                for (const result of searchResults) {
                    dbContext += `\n【${result.semester} - ${result.mapel}】\n`;
                    if (result.materi.length > 0) {
                        dbContext += `  Materi: ${result.materi.join(", ")}\n`;
                    }
                    if (result.tugas.length > 0) {
                        dbContext += `  Tugas: ${result.tugas.join(", ")}\n`;
                    }
                }
            } else {
                dbContext = "\n\n[DATA DATABASE BOT]\n" + database.getInfo();
            }
        }
        
        if (!GROQ_API_KEY) {
            if (isAskingDatabase) {
                return database.getInfo();
            }
            return "❌ AI tidak aktif. Ketik .menumapel untuk fitur lainnya.";
        }
        
        try {
            const res = await axios.post(
                "https://api.groq.com/openai/v1/chat/completions",
                {
                    model: "llama-3.3-70b-versatile",
                    messages: [
                        {
                            role: "system",
                            content: "Kamu guru SMK TKJ. Jawab singkat, jelas, tidak bertele-tele. Maksimal 2 kalimat."
                        },
                        {
                            role: "user",
                            content: `Pertanyaan: ${question}\n\n${dbContext}\n\nJawab:`
                        }
                    ],
                    max_tokens: 200,
                    temperature: 0.5
                },
                {
                    headers: {
                        Authorization: `Bearer ${GROQ_API_KEY}`,
                        "Content-Type": "application/json"
                    },
                    timeout: 30000
                }
            );

            let answer = res.data.choices[0].message.content;
            
            if (answer.length > 500) {
                answer = answer.substring(0, 500) + "...";
            }
            
            return answer;
        } catch (err) {
            console.error("AI Error:", err.response?.data || err.message);
            
            if (isAskingDatabase) {
                return database.getInfo();
            }
            
            return "❌ AI error. Coba lagi nanti.";
        }
    }
}

// ================= COMMAND HANDLER =================
class CommandHandler {
    constructor(sock, database) {
        this.sock = sock;
        this.db = database;
    }

    async deleteTugas(jid, semester, args) {
    // Gabungkan semua args karena mapel bisa lebih dari 1 kata
    const fullArg = args.join(" ");
    const parts = fullArg.split(" ");
    
    if (parts.length !== 2) {
        await this.sock.sendMessage(jid, {
            text: `⚠️ *Format Salah!*\n\nFormat: .${semester === "semester1" ? "-tugass1" : "-tugass2"} <mapel>|<urutan>\n\nContoh:\n.${semester === "semester1" ? "-tugass1" : "-tugass2"} tkj 2\n\n💡 Tips: Gunakan .listtugass1 atau .listtugass2 untuk melihat urutan tugas`
        });
        return;
    }
    
    const mapel = parts[0].trim().toLowerCase();
    const indexStr = parts[1].trim();
    const index = parseInt(indexStr);
    
    // Validasi mapel
    if (!this.db.db[semester][mapel]) {
        await this.sock.sendMessage(jid, {
            text: `❌ Mata pelajaran "${mapel}" tidak ditemukan di ${semester}.\n\n📋 Daftar mapel: ${this.db.getAllMapel(semester).join(", ")}`
        });
        return;
    }
    
    // Validasi index
    if (isNaN(index) || index <= 0) {
        await this.sock.sendMessage(jid, {
            text: "❌ Urutan harus berupa angka positif!\n\nContoh: .-tugass1 tkj|2"
        });
        return;
    }
    
    const target = this.db.getMapel(semester, mapel);
    const tugasList = target.tugas;
    
    // Validasi apakah urutan tersedia
    if (index - 1 >= tugasList.length) {
        await this.sock.sendMessage(jid, {
            text: `❌ Urutan ${index} tidak ditemukan!\n\n📝 ${mapel} hanya memiliki ${tugasList.length} tugas.\n\nGunakan .listtugass1 untuk melihat semua tugas beserta urutannya.`
        });
        return;
    }
    
    // Hapus tugas
    const removed = tugasList.splice(index - 1, 1);
    this.db.save();
    
    // Kirim konfirmasi dengan informasi lengkap
    let response = `✅ *Tugas Berhasil Dihapus!*\n\n`;
    response += `📚 Semester: ${semester === "semester1" ? "Semester 1" : "Semester 2"}\n`;
    response += `📘 Mata Pelajaran: ${mapel}\n`;
    response += `📝 Tugas yang dihapus: "${removed[0]}"\n`;
    response += `🔢 Urutan: ${index}\n\n`;
    
    if (tugasList.length > 0) {
        response += `📋 *Sisa Tugas:*\n`;
        tugasList.forEach((t, i) => {
            response += `   ${i + 1}. ${t}\n`;
        });
    } else {
        response += `✨ Sekarang tidak ada tugas untuk ${mapel}`;
    }
    
    await this.sock.sendMessage(jid, { text: response });
}

    async handle(jid, cmd, args) {
        const ownerOnlyCommands = [
            "+materis1", "+materis2", 
            "+tugass1", "+tugass2",
            "-materis1", "-materis2",
            "-tugass1", "-tugass2",
            "addowner", "removeowner", "listowner"
        ];
        
        if (ownerOnlyCommands.includes(cmd)) {
            if (!isOwner(jid)) {
                await this.sock.sendMessage(jid, {
                    text: `❌ *Akses Ditolak!*\n\nPerintah "${cmd}" hanya bisa digunakan oleh Owner Bot.\n\nKetik .cekjid untuk lihat JID Anda.`
                });
                return true;
            }
        }

        const commands = {
            "menumapel": () => this.showMenu(jid),
            "cekjid": () => this.cekJid(jid),
            "semester1": () => this.handleSemester(jid, "semester1", args),
            "semester2": () => this.handleSemester(jid, "semester2", args),
            "img": () => this.handleImage(jid, args),
            "caridb": () => this.handleSearch(jid, args),
            "listmateris1": () => this.listMateri(jid, "semester1"),
            "listmateris2": () => this.listMateri(jid, "semester2"),
            "listtugass1": () => this.listTugas(jid, "semester1"),
            "listtugass2": () => this.listTugas(jid, "semester2"),
            "+materis1": () => this.addMateri(jid, "semester1", args),
            "+materis2": () => this.addMateri(jid, "semester2", args),
            "+tugass1": () => this.addTugas(jid, "semester1", args),
            "+tugass2": () => this.addTugas(jid, "semester2", args),
            "-materis1": () => this.deleteMateri(jid, "semester1", args),
            "-materis2": () => this.deleteMateri(jid, "semester2", args),
            "-tugass1": () => this.deleteTugas(jid, "semester1", args),
            "-tugass2": () => this.deleteTugas(jid, "semester2", args),
            "addowner": () => this.addOwner(jid, args),
            "removeowner": () => this.removeOwner(jid, args),
            "listowner": () => this.listOwner(jid),
            "tanya": () => this.handleAI(jid, args),
        };

        const handler = commands[cmd];
        if (handler) {
            await handler();
            return true;
        }
        return false;
    }

                text: `❌ Format JID tidak valid!\n\nJID harus berisi '@'\nContoh: 6285399180750@s.whatsapp.net`

        if (addOwner(newOwnerJid)) {

    async handleSemester(jid, semester, args) {
        if (args.length === 0) {
            let out = `📚 ${semester.toUpperCase()}\n${line()}\n`;
            for (const m of this.db.getAllMapel(semester)) {
                out += `▸ ${m}\n`;
            }
            await this.sock.sendMessage(jid, { text: out });
            return;
        }

        const mapel = args[0].toLowerCase();
        const data = this.db.db[semester][mapel];
        
        if (!data) {
            await this.sock.sendMessage(jid, {
                text: `❌ Mata pelajaran "${mapel}" tidak ditemukan.\n\n📋 Daftar mapel: ${this.db.getAllMapel(semester).join(", ")}`

            return;

        let out = `📘 *${mapel.toUpperCase()}* - ${semester.toUpperCase()}\n${line()}\n\n`;
        
        out += `📖 *MATERI:*\n`;
        if (data.materi.length === 0) {
            out += `   - Belum ada materi\n`;
        } else {
            data.materi.forEach((m, i) => {
                out += `   ${i + 1}. ${m}\n`;
            });
        }
        
        out += `\n📝 *TUGAS:*\n`;
        if (data.tugas.length === 0) {
            out += `   - Belum ada tugas\n`;
        } else {
            data.tugas.forEach((t, i) => {
                out += `   ${i + 1}. ${t}\n`;
            });
        }
        
        await this.sock.sendMessage(jid, { text: out });
    }

    async handleImage(jid, args) {
        const mapel = args[0]?.toLowerCase();
        const semester = args[1]?.toLowerCase();
        
        if (!mapel || !semester) {
            await this.sock.sendMessage(jid, {
                text: "Format: .img <mapel> <semester1/semester2>\nContoh: .img tkj semester1"
            });
            return;
        }
        
        if (semester !== "semester1" && semester !== "semester2") {
            await this.sock.sendMessage(jid, {
                text: "Semester harus 'semester1' atau 'semester2'"
            });
            return;
        }
        
        const data = this.db.db[semester][mapel];
        if (!data) {
            await this.sock.sendMessage(jid, {
                text: `❌ Mata pelajaran "${mapel}" tidak ditemukan di ${semester}`
            });
            return;
        }
        
        try {
            await this.sock.sendMessage(jid, { text: "🖼️ Sedang membuat gambar..." });
            const imageBuffer = await ImageGenerator.generateMapelImage(mapel, semester, data);
            await this.sock.sendMessage(jid, {
                image: imageBuffer,
                caption: `📘 *${mapel.toUpperCase()}* - ${semester.toUpperCase()}\n\n📖 Materi: ${data.materi.length} item\n📝 Tugas: ${data.tugas.length} item`
            });
        } catch (err) {
            console.error(err);
            await this.sock.sendMessage(jid, {
                text: "❌ Gagal membuat gambar."
            });
        }
    }

    async handleSearch(jid, args) {
        const keyword = args.join(" ");
        if (!keyword) {
            await this.sock.sendMessage(jid, {
                text: "Contoh: .caridb jaringan"
            });
            return;
        }
        
        const results = this.db.search(keyword);
        
        if (results.length === 0) {
            await this.sock.sendMessage(jid, {
                text: `🔍 Tidak ditemukan hasil untuk "${keyword}"`
            });
            return;
        }
        
        let out = `🔍 HASIL PENCARIAN: "${keyword}"\n${line()}\n\n`;
        for (const result of results) {
            out += `📚 ${result.semester} - ${result.mapel}\n`;
            if (result.materi.length > 0) {
                out += `   📖 Materi:\n`;
                result.materi.forEach(m => out += `      • ${m}\n`);
            }
            if (result.tugas.length > 0) {
                out += `   📝 Tugas:\n`;
                result.tugas.forEach(t => out += `      • ${t}\n`);
            }
            out += `\n`;
        }
        
        await this.sock.sendMessage(jid, { text: out });
    }

    async listMateri(jid, semester) {
        let out = `📖 MATERI ${semester.toUpperCase()}\n${line()}\n\n`;
        for (const m in this.db.db[semester]) {
            out += `📘 ${m}\n`;
            if (!this.db.db[semester][m].materi.length) {
                out += "   - kosong\n\n";
                continue;
            }
            this.db.db[semester][m].materi.forEach((x, i) => {
                out += `   ${i + 1}. ${x}\n`;
            });
            out += "\n";
        }
        await this.sock.sendMessage(jid, { text: out });
    }

    async listTugas(jid, semester) {
        let out = `📝 TUGAS ${semester.toUpperCase()}\n${line()}\n\n`;
        for (const m in this.db.db[semester]) {
            out += `📌 ${m}\n`;
            if (!this.db.db[semester][m].tugas.length) {
                out += "   - kosong\n\n";
                continue;
            }
            this.db.db[semester][m].tugas.forEach((x, i) => {
                out += `   ${i + 1}. ${x}\n`;
            });
            out += "\n";
        }
        await this.sock.sendMessage(jid, { text: out });
    }

    async addMateri(jid, semester, args) {
        const mapel = args[0]?.toLowerCase();
        const judul = args.slice(1).join(" ");

        if (!mapel || !judul) {
            await this.sock.sendMessage(jid, {
                text: `Format: .${semester === "semester1" ? "+materis1" : "+materis2"} mapel judul materi`
            });
            return;
        }

        this.db.getMapel(semester, mapel).materi.push(judul);
        this.db.save();

        await this.sock.sendMessage(jid, {
            text: `✅ Materi tersimpan di ${semester}/${mapel}`
        });
    }

    async addTugas(jid, semester, args) {
        const mapel = args[0]?.toLowerCase();
        const judul = args.slice(1).join(" ");

        if (!mapel || !judul) {
            await this.sock.sendMessage(jid, {
                text: `Format: .${semester === "semester1" ? "+tugass1" : "+tugass2"} mapel judul tugas`
            });
            return;
        }

        this.db.getMapel(semester, mapel).tugas.push(judul);
        this.db.save();

        await this.sock.sendMessage(jid, {
            text: `✅ Tugas tersimpan di ${semester}/${mapel}`
        });
    }

    async deleteMateri(jid, semester, args) {
        const mapel = args[0]?.toLowerCase();
        const index = parseInt(args[1]);
        const target = this.db.getMapel(semester, mapel);

        if (!target.materi[index - 1]) {
            await this.sock.sendMessage(jid, {
                text: "Urutan tidak valid"
            });
            return;
        }

        const removed = target.materi.splice(index - 1, 1);
        this.db.save();

        await this.sock.sendMessage(jid, {
            text: `🗑️ Dihapus: ${removed[0]}`
        });
    }

    async handleAI(jid, args) {
        const question = args.join(" ");
        if (!question) {
            await this.sock.sendMessage(jid, {
                text: "Contoh: .tanya apa itu subnetting\n\nAtau tanya database:\n.tanya materi TKJ semester 1?"
            });
            return;
        }

        const answer = await AIService.ask(question, this.db);
        await this.sock.sendMessage(jid, { text: `🤖 ${answer}` });
    }
}

// ================= MAIN BOT =================
class Bot {
    constructor() {
        this.sock = null;
        this.db = new Database();
    }

    async start() {
        const { state, saveCreds } = await useMultiFileAuthState("./auth");

        this.sock = makeWASocket({
            auth: state,
            logger: pino({ level: "silent" }),
            printQRInTerminal: true
        });

        this.sock.ev.on("creds.update", saveCreds);
        this.sock.ev.on("connection.update", this.handleConnection.bind(this));
        this.sock.ev.on("messages.upsert", this.handleMessages.bind(this));

        console.log("🚀 BOT READY - AI bisa membaca database!");
        console.log("📁 Database tersimpan di database.json");
        console.log("👑 Jumlah Owner:", OWNER_JIDS.length);
        if (OWNER_JIDS.length > 0) {
            console.log("👑 Owner JIDs:", OWNER_JIDS);
        }
    }

    handleConnection({ connection, qr, lastDisconnect }) {
        if (qr) {
            console.log("SCAN QR");
            qrcode.generate(qr, { small: true });
        }

        if (connection === "open") {
            console.log("BOT CONNECTED");
        }

        if (connection === "close") {
            const code = lastDisconnect?.error?.output?.statusCode;
            if (code !== DisconnectReason.loggedOut) {
                this.start();
            }
        }
    }

    async handleMessages({ messages }) {
        const commandHandler = new CommandHandler(this.sock, this.db);

        for (const msg of messages) {
            const jid = msg.key.remoteJid;
            const text = this.extractText(msg);
            
            if (text && text.startsWith(".")) {
                console.log(`📨 Command dari JID: ${jid} | Pesan: ${text}`);
            }
            
            if (!text || !text.startsWith(".")) continue;

            const raw = text.slice(1).trim();
            const cmd = raw.split(" ")[0].toLowerCase();
            const args = raw.split(" ").slice(1);

            await commandHandler.handle(jid, cmd, args);
        }
    }

    extractText(msg) {
        return msg.message?.conversation ||
               msg.message?.extendedTextMessage?.text ||
               msg.message?.imageMessage?.caption ||
               msg.message?.videoMessage?.caption ||
               "";
    }
}

// ================= RUN BOT =================
const bot = new Bot();
bot.start();
